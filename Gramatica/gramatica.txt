

S -> tipo NOMBRE PARENTESIS_IZQ PARAMS PARENTESIS_DER LLAVE_IZQ L LLAVE_DER

NOMBRE -> IDENTIFICADOR
NOMBRE -> PRINCIPAL

tipo -> TIPO_ENTERO
tipo -> TIPO_DECIMAL
tipo -> TIPO_CARACTER
tipo -> TIPO_BOOLEANO
tipo -> TIPO_CADENA
tipo -> VOID

PARAMS -> tipo IDENTIFICADOR PARAMS_TAIL
PARAMS -> ''

PARAMS_TAIL -> COMA tipo IDENTIFICADOR PARAMS_TAIL
PARAMS_TAIL -> ''

L -> I L
L -> ''

I -> SALIDA MENOR_QUE MENOR_QUE VALOR SALTO PUNTO_COMA
I -> IDENTIFICADOR IGUAL E PUNTO_COMA
I -> IDENTIFICADOR PARENTESIS_IZQ PARENTESIS_DER PUNTO_COMA
I -> MIENTRAS PARENTESIS_IZQ C PARENTESIS_DER LLAVE_IZQ L LLAVE_DER
I -> PARA PARENTESIS_IZQ ASIG PUNTO_COMA C PUNTO_COMA ASIG PARENTESIS_DER LLAVE_IZQ L LLAVE_DER
I -> SI PARENTESIS_IZQ C PARENTESIS_DER LLAVE_IZQ L LLAVE_DER SINO_PARTE
I -> REGRESAR E PUNTO_COMA

VALOR -> IDENTIFICADOR
VALOR -> CADENA

SALTO -> MENOR_QUE MENOR_QUE FIN_LINEA
SALTO -> ''

ASIG -> IDENTIFICADOR IGUAL E

SINO_PARTE -> SINO LLAVE_IZQ L LLAVE_DER
SINO_PARTE -> ''

C -> NEGACION C
C -> E OP E C_PRIME

C_PRIME -> Y_LOGICO E OP E C_PRIME
C_PRIME -> O_LOGICO E OP E C_PRIME
C_PRIME -> ''

OP -> MENOR_QUE
OP -> MAYOR_QUE
OP -> IGUAL_IGUAL
OP -> DIFERENTE
OP -> MENOR_IGUAL
OP -> MAYOR_IGUAL

E -> R E_PRIME
E_PRIME -> SUMA R E_PRIME
E_PRIME -> RESTA R E_PRIME
E_PRIME -> ''

R -> F R_PRIME
R_PRIME -> MULTIPLICACION F R_PRIME
R_PRIME -> DIVISION F R_PRIME
R_PRIME -> MODULO F R_PRIME
R_PRIME -> ''

F -> PARENTESIS_IZQ E PARENTESIS_DER
F -> IDENTIFICADOR
F -> ENTERO
F -> DECIMAL






S -> tipo id ( PARAMS ) { L }

PARAMS -> tipo id PARAMS_TAIL
PARAMS -> ''

PARAMS_TAIL -> , tipo id PARAMS_TAIL
PARAMS_TAIL -> ''

tipo -> entero
tipo -> decimal
tipo -> caracter
tipo -> booleano
tipo -> cadena
tipo -> void

L -> I L
L -> ''

I -> salida << id SALTO ;
I -> id = E ;
I -> mientras ( C ) { L }
I -> para ( ASIG ; C ; ASIG ) { L }
I -> si ( C ) { L } SINO_PARTE
I -> regresar E ;

SALTO -> << fin_linea
SALTO -> ''

ASIG -> id = E

SINO_PARTE -> sino { L }
SINO_PARTE -> ''

C -> E OP E C'
C' -> && E OP E C'
C' -> || E OP E C'
C' -> ''

OP -> <
OP -> >
OP -> ==
OP -> !=
OP -> <=
OP -> >=

E -> T E'
E' -> + T E'
E' -> - T E'
E' -> ''
T -> id
T -> num
